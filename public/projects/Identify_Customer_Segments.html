# Function to convert strings into lists
def missing_parser(encode_str):
    if pd.isna(encode_str) or str(encode_str).strip() in ("", "[]"):
        return []
    
    encode_str = str(encode_str).strip()
    
    if encode_str.startswith('[') and encode_str.endswith(']'):
        encode_str = encode_str[1:-1]
    
    new_items = []
    
    for item in encode_str.split(','):
        i = item.strip().strip("'").strip('"')
        if i == "":
            continue
        try:
            new_items.append(int(i))
        except ValueError:
            new_items.append(i)
    
    return new_items

# Function to determine if there are unique integer values and turns them categorical in nature
def is_discrete(n, max_unique=20):
    if not pd.api.types.is_numeric_dtype(n):
        return False
    
    values = n.dropna()
    
    if values.empty:
        return False
    
    integers = np.all(np.isclose(values, np.round(values)))
    return integers and values.nunique(dropna=True) <= max_unique

# Function to compare distributions with all proper categorical data or histogram if data is all numerical
def compare_distros(col, low, high, top_k=12):
    x_low = low[col].dropna()
    x_high = high[col].dropna()
    
    fig, axes = plt.subplots(1, 2, figsize=(12, 4))
    fig.suptitle(f"{col}, Low vs High Missing")
    
    if is_discrete(x_low) or (not pd.api.types.is_numeric_dtype(x_low)):
        low_n = x_low.value_counts(normalize=True).head(top_k)
        high_n = x_high.value_counts(normalize=True).reindex(low_n.index).fillna(0.0)
        
        sns.barplot(x=low_n.index, y=low_n.values, ax=axes[0])
        axes[0].set_title("Low Missing"); axes[0].set_xlabel(col); axes[0].set_ylabel("Proportion"); axes[0].tick_params(axis='x', rotation=45)

# Function to map the remaining binaries to {0, 1} for consistency
def update_map(df, col):
    if col not in df.columns:
        return
    
    c = df[col]
    v = sorted(pd.unique(c.dropna()))
    
    if set(v) == {0.0, 1.0} or set(v) == {0, 1}:
        df[col] = c.astype("float32", copy=False)
        return
    
    if len(v) != 2:
        return
    
    v0, v1 = v
    mapping = {v0: 0.0, v1: 1.0}
    
    df[col] = c.map(mapping).astype('float32')

import ast

def build_missing_map(feature_summary):
    fs = feature_summary.copy()
    fs = fs["missing_or_unknown"].astype(str).str.strip() != ""
    fs = feature_summary.loc[fs]
    missing = {}
    for _, r in fs.iterrows():
        col = r["attribute"]
        raw = str(r["missing_or_unknown"]).strip()
        try:
            code = ast.literal_eval(raw)
            if not isinstance(code, (list, tuple)): code = [code]
        except Exception:
            token = [t.strip().strip("'").strip('"')
                     for t in raw.strip("[]").split(",") if t.strip()]
            code = []
            for t in token:
                try:
                    f = float(t); code.append(int(f) if f.is_integer() else f)
                except Exception:
                    code.append(t)
        missing[col] = code
    return missing

missing_map = build_missing_map(feature_summary)

def clean_data(df, feature_summary, missing_map):
    output = df.copy()

    mm = missing_map if missing_map is not None else build_missing_map(feature_summary)
    for col, code in mm.items():
        if col not in output.columns:
            continue

        # flatten codes
        flattened = []
        for c in (code if isinstance(code, (list, tuple, set)) else [code]):
            if isinstance(c, (list, tuple, np.ndarray, set)):
                flattened.extend(list(c))
            else:
                flattened.append(c)

        # split string vs numeric
        str_codes = {str(c).strip().upper() for c in flattened if isinstance(c, str)}
        num_codes = [c for c in flattened if not isinstance(c, str)]

        s = output[col]

        if str_codes:
            su = s.astype(str).str.strip().str.upper()
            s = s.mask(su.isin(str_codes), np.nan)

        if num_codes:
            sn = pd.to_numeric(s, errors="ignore")
            s = sn.mask(pd.Series(sn).isin(num_codes).values, np.nan)

        output[col] = s

    # Drops
    high_missing = ['TITEL_KZ','AGER_TYP','KK_KUNDENTYP','KBA05_BAUMAX','GEBURTSJAHR','ALTER_HH']
    output = output.drop(columns=[col for col in high_missing if col in output.columns], errors="ignore")

    multi_drop = ["CJT_GESAMTTYP","FINANZTYP","ZABEOTYP","GEBAEUDETYP",
                  "CAMEO_DEUG_2015","CAMEO_DEU_2015","GFK_URLAUBERTYP",

            normal = s.astype(str).str.strip().str.upper()

                mapped.loc[rem] = coerced.where(coerced.isin([0, 1]), other=np.nan)
            output["OST_WEST_KZ"] = mapped.astype("float32")
        else:

            s = s.astype(str).str.strip().str.upper().map({"O": 0.0, "W": 1.0})
        output[col] = pd.to_numeric(s, errors="coerce").astype("float32")

    # PJ_*
    if ("PRAEGENDE_JUGENDJAHRE" in output.columns) and (

        output["PJ_MOVEMENT"] = movement

    if ("CAMEO_INTL_2015" in output.columns) and (
        ("CAMEO_WEALTH" not in output.columns) or ("CAMEO_LIFESTAGE" not in output.columns)
    ):
        s = output["CAMEO_INTL_2015"].astype(str).str.strip().str.upper()

    fs = feature_summary.copy()

    kept_mixed  = [col for col in ["WOHNLAGE","PLZ8_BAUMAX"] if col in present]
