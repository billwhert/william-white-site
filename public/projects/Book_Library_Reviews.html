base = "https://openlibrary.org/search.json"

def ol_search(title, author=None, limit=5):
    # search open library for a book by title (and optionally author). Returns top N results as a list of dicts.
    params = {
        "q": title,
        "limit": limit
    }

    if author:
        params["author"] = author

    r = requests.get(base, params=params, timeout=20)
    r.raise_for_status()
    return r.json().get("docs", [])

# Normalization helpers (for robust joins on text)

def normalize_text(s:str):
    # Lowercase, strip, remove non-alphanumerics to only keep a-z and 0-9
    if pd.isna(s):
        return ""

    s = str(s).lower().strip()
    
    return re.sub(r"[^a-z0-9]+", "", s)

def make_title_key(title: str):
    # Create a normalized key for titles.
    return normalize_text(title)

def make_author_key(authors: str):
    # Create a normalized key from the first listed author.
    # df_ratings 'authors' often has 'Author A, Author B'
    first = str(authors).split(",")[0]
    return normalize_text(first)

# Selector for Open Library search results
def pick_best_openlibrary(doc_list, want_title_key: str, want_author_key: str | None):
    '''Given a list of OL docs, choose the 'best' one for title/author. Heuristics (in order):
    1) exact title_key match (normalized) if possible
    2) prefer matching author_key when available
    3) among remaining, pick highest 'edition_count'
    Returns a dict with the core fields we care about, or None.
    '''

    if not doc_list:
        return None

    #Build candidates with normlaized keys
    cands = []
    for d in doc_list:
        t_raw = d.get("title", "")
        a_raw = (d.get("author_name")) or [""][0]
        t_key = make_title_key(t_raw)
        a_key = make_author_key(a_raw)
        cands.append((d, t_key, a_key))

    # 1) filter by exact title match, else fall back to all
    exact_title = [c for c in cands if c[1] ==  want_title_key]
    pool = exact_title if exact_title else cands

    # 2) if author provided, prefer matches
    if want_author_key:
        author_matches = [c for c in pool if c[2] == want_author_key]
        if author_matches:
            pool = author_matches

    # 3) pick highest edition_count as a proxy for prominence
    def edcount(x):
        d = x[0]
        return d.get("edition_count") or 0

    best = max(pool, key=edcount)[0]

    #return compact fields we'll use downstream
    return {
        'ol_work_key': best.get('key'),
        'title': best.get('title'),
        'first_publish_year': best.get('first_publish_year'),
        'edition_count': best.get('edition_count'),
        'has_fulltext': bool(best.get("has_fulltext")),
        'language_count': len(best.get('language') or []),
        'author_name': (best.get('author_name') or [""][0])
    }

# Lightwight cache to avoid repeating API calls with reruns

ol_cache = {}

def fetch_openlibrary_best(title: str, authors: str | None, limit=5, sleep=.15):
    '''Wrapper that:
    1) builds normalized keys from our ratings row
    2) calls Open Library search
    3) picks the best doc with our heuristics
    4) caches the result
    Returns a compact dict (or None)'''

    t_key = make_title_key(title)
    a_key = make_author_key(authors)
    cache_key = (t_key, a_key)

    if cache_key in ol_cache:
        return ol_cache[cache_key]

    docs = ol_search(title, author=authors, limit=limit)
    best = pick_best_openlibrary(docs, want_title_key=t_key, want_author_key=a_key)
    ol_cache[cache_key] = best
    time.sleep(sleep) #polite to avoid hitting the API repeatedly with thousands of items
    return best

Outcome:

The relationship between book popularity and average quality appears weak. Across our sample, the Pearson and Spearman correlations between ratings_count and average_rating are ~0, and the scatter (log-x) shows wide vertical spread at all popularity levels. Likewise, edition_count has near-zero correlation with average_rating; the bucketed means differ by ~0.1 and their 95% CIs largely overlap. These results suggest that highly popular or long-published titles are not consistently higher-rated. Caveats include small n in some buckets and edition_count being only a proxy for longevity; matching by title/author can also miss edge cases.
